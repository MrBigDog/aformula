40 bool Parser::setVariable (const std::string &name, double *pointer)
41 {
42  // If this name is already some kind of identifier, you can't
43  // register such a variable
44  int type = getIdentifierType (name);
45 
46  if (type != 0)
47  {
48  std::string str;
49 
50  if (type == TOKEN_IDENTIFIER_FUNCTION)
51  str = "function";
52  else if (type == TOKEN_IDENTIFIER_CONSTANT)
53  str = "mathematical constant";
54  else
55  str = "variable";
56 
57  error (boost::format ("Cannot register variable %1%, "
58  "a %2% with this name already exists!") %
59  name % str);
60  return false;
61  }
62 
63  Variable var;
64 
65  var.name = name;
66  var.pointer = pointer;
67 
68  variables.push_back (var);
69  return true;
70 }

92 std::string Parser::formatToken () const
93 {
94  const char *tokenStrings[] =
95  {
96  "an invalid token",
97  "end-of-formula",
98  "a function name",
99  "a constant name",
100  "a variable name",
101  "an operator",
102  "a number",
103  "an open parenthesis",
104  "a close parenthesis",
105  "a comma"
106  };
107 
108  // Mark up identifiers and operators with their corresponding string
109  if (currentToken >= TOKEN_IDENTIFIER_FUNCTION &&
110  currentToken <= TOKEN_OPERATOR)
111  return boost::str (boost::format ("%1% (\"%2%\")") %
112  tokenStrings[currentToken] % strToken);
113 
114  // Mark up numbers with their number
115  if (currentToken == TOKEN_NUMBER)
116  return boost::str (boost::format ("%1% (%2%)") %
117  tokenStrings[currentToken] % numToken);
118 
119  // Everything else is just the token name
120  return std::string (tokenStrings[currentToken]);
121 }

177 int Parser::getToken ()
178 {
179  // Eat whitespace
180  while (isspace (parseBuffer[0]))
181  parseBuffer.erase (0, 1);
182 
183  // Bail if we're done
184  if (!parseBuffer.length ())
185  return TOKEN_END;
186 
187  // See if the first part of the parseBuffer is an operator string
188  for (int i = 0 ; i < NUM_OPERATORS ; i++)
189  {
190  if (parseBuffer.compare (0, strlen (operators[i].name), operators[i].name) == 0)
191  {
192  // Indeed, an operator is here
193  strToken.erase ();
194 
195  strToken = operators[i].name;
196  parseBuffer.erase (0, strlen (operators[i].name));
197 
198  return TOKEN_OPERATOR;
199  }
200  }
201 
202  // If the next thing begins with a letter, it's some kind of
203  // identifier
204  if (isalpha (parseBuffer[0]))
205  {
206  strToken.erase ();
207 
208  while (parseBuffer.length () &&
209  (isalpha (parseBuffer[0]) || isdigit (parseBuffer[0])))
210  {
211  strToken += parseBuffer[0];
212  parseBuffer.erase (0,1);
213  }
214 
215  return getIdentifierType (strToken);
216  }
217 
218  // Neither operator nor identifier, try number
219  bool haveDot = false;
220  if (isdigit (parseBuffer[0]) || parseBuffer[0] == '.')
221  {
222  strToken.erase ();
223 
224  while (parseBuffer.length () &&
225  (isdigit (parseBuffer[0]) || parseBuffer[0] == '.'))
226  {
227  strToken += parseBuffer[0];
228 
229  // Two decimal points?
230  if (parseBuffer[0] == '.' && haveDot)
231  return TOKEN_BAD;
232  if (parseBuffer[0] == '.')
233  haveDot = true;
234 
235  parseBuffer.erase (0, 1);
236  }
237 
238  // Convert the string token to a decimal
239  const char *str = strToken.c_str ();
240  char *endptr;
241  numToken = strtod (str, &endptr);
242 
243  // Check for failure
244  if (numToken == 0.0 && endptr == str)
245  return TOKEN_BAD;
246 
247  return TOKEN_NUMBER;
248  }
249 
250  // Syntactic glue that doesn't parse into the tree
251  if (parseBuffer[0] == '(')
252  {
253  strToken = parseBuffer[0];
254  parseBuffer.erase (0,1);
255  return TOKEN_PAREN_OPEN;
256  }
257 
258  if (parseBuffer[0] == ')')
259  {
260  strToken = parseBuffer[0];
261  parseBuffer.erase (0,1);
262  return TOKEN_PAREN_CLOSE;
263  }
264 
265  if (parseBuffer[0] == ',')
266  {
267  strToken = parseBuffer[0];
268  parseBuffer.erase (0,1);
269  return TOKEN_COMMA;
270  }
271 
272  // What could that be?
273  return TOKEN_BAD;
274 }

345 ExprAST *Parser::parseFunctionIdentifierExpr ()
346 {
347  std::string function (strToken);
348  getNextToken ();
349 
350  if (currentToken != TOKEN_PAREN_OPEN)
351  {
352  error (boost::format ("Call to function %1% without following opening parenthesis!")
353  % function);
354  return NULL;
355  }
356  getNextToken ();
357 
358  std::vector<ExprAST *> args;
359  if (currentToken != TOKEN_PAREN_CLOSE)
360  {
361  while (1)
362  {
363  ExprAST *arg = parseExpression ();
364  if (!arg)
365  return NULL;
366  args.push_back (arg);
367 
368  if (currentToken == TOKEN_PAREN_CLOSE)
369  break;
370  if (currentToken != TOKEN_COMMA)
371  {
372  error (boost::format ("Expected ')' or ',' in argument list to function %1%!")
373  % function);
374  return NULL;
375  }
376  getNextToken ();
377  }
378  }
379 
380  // Look up expected number of arguments
381  for (int i = 0 ; i < NUM_FUNCTIONS ; i++)
382  {
383  if (function == functions[i].name)
384  {
385  if (args.size () < functions[i].numArgs)
386  {
387  error (boost::format ("Not enough arguments for function %1% (%2%, expecting %3%)!")
388  % function % args.size () % functions[i].numArgs);
389  return NULL;
390  }
391  else if (args.size () > functions[i].numArgs)
392  {
393  error (boost::format ("Too many arguments for function %1% (%2%, expecting %3%)!")
394  % function % args.size () % functions[i].numArgs);
395  return NULL;
396  }
397  }
398  }
399 
400  // Eat the ')'
401  getNextToken ();
402 
403  return new CallExprAST (function, args);
404 }

474 ExprAST *Parser::parsePrimary ()
475 {
476  // At the top level, we have to have either an identifier of some kind,
477  // a number, or an open paren. We might also have a unary minus sign, handle
478  // that specially right here.
479  switch (currentToken)
480  {
481  case TOKEN_IDENTIFIER_FUNCTION:
482  return parseFunctionIdentifierExpr ();
483  case TOKEN_IDENTIFIER_CONSTANT:
484  return parseConstantIdentifierExpr ();
485  case TOKEN_IDENTIFIER_VARIABLE:
486  return parseVariableIdentifierExpr ();
487  case TOKEN_NUMBER:
488  return parseNumberExpr ();
489  case TOKEN_PAREN_OPEN:
490  return parseOpenParenExpr ();
491  case TOKEN_OPERATOR:
492  if (strToken == "-")
493  return parseUnaryMinusExpr ();
494  // Fall through for any other operator
495  default:
496  error (boost::format ("Expecting an expression, got %1% instead!") %
497  formatToken ());
498  return NULL;
499  }
500 }

502 ExprAST *Parser::parseBinOpRHS (int exprPrecedence, ExprAST *LHS)
503 {
504  while (1)
505  {
506  // This should be either an operator, a close-paren, comma,
507  // or the end of the formula.
508  if (currentToken != TOKEN_OPERATOR &&
509  currentToken != TOKEN_PAREN_CLOSE &&
510  currentToken != TOKEN_COMMA &&
511  currentToken != TOKEN_END)
512  {
513  error (boost::format ("Attempting to parse the RHS of a binary operator, "
514  "and got %1%!") % formatToken ());
515  return NULL;
516  }
517 
518  // Get the precedence
519  int tokenPrecedence = -1;
520 
521  for (int i = 0 ; i < NUM_OPERATORS ; i++)
522  {
523  if (strToken == operators[i].name)
524  {
525  tokenPrecedence = operators[i].precedence;
526  break;
527  }
528  }
529 
530  // We're done if we run into an equivalent or lower-precedence binary
531  // operator (or, notably, something that's not an operator at all, like
532  // the end of the formula)
533  if (tokenPrecedence < exprPrecedence)
534  return LHS;
535 
536  // Pull off the operator
537  std::string binOp = strToken;
538  getNextToken ();
539 
540  // Parse the primary after the operator
541  ExprAST *RHS = parsePrimary ();
542  if (!RHS)
543  return NULL;
544 
545  // We've now parsed a full binary operator pair, "A op B". If the next
546  // thing is a binary operator (which it should be, or it should be
547  // nothing at all), we need to see whether we want "(A op B) op2 ..." or
548  // "A op (B op2 ...)" by precedence rules. Look ahead:
549  int nextPrecedence = -1;
550  for (int i = 0 ; i < NUM_OPERATORS ; i++)
551  {
552  if (strToken == operators[i].name)
553  {
554  nextPrecedence = operators[i].precedence;
555  break;
556  }
557  }
558 
559  if (tokenPrecedence < nextPrecedence)
560  {
561  // Next precedence is greater, we want "A op (B op2 ...)". Do that by
562  // parsing B op2 ... into a new RHS.
563  RHS = parseBinOpRHS (tokenPrecedence + 1, RHS);
564  if (RHS == NULL)
565  return NULL;
566  }
567 
568  // Next precedence is lower, we want "(A op B) op2 ...", so make a new LHS
569  LHS = new BinaryExprAST (binOp, LHS, RHS);
570 
571  // And loop back to the beginning
572  }
573 }
