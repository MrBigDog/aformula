59 LLVMFormula::LLVMFormula ()
60 {
61  // We need to initialize the native target info, once.
62  static bool initializeNative = false;
63  if (!initializeNative)
64  {
65  InitializeNativeTarget ();
66  initializeNative = true;
67  }
68 
69  // Build a module and a JIT engine
70  module = llvm::make_unique<Module>("AFormula JIT", globalContext);
71 
72  // The Engine is going to take control of this module,
73  // don't delete them later.
74  std::string errorString;
75  engine = EngineBuilder (std::move(module)).setErrorStr (&errorString).create ();
76  if (!engine)
77  {
78  // We won't let you call buildFunction if you catch this error
79  errorMessage.reset (new std::string ("LLVM Error: " + errorString));
80  return;
81  }
82 
83  // Create an IRBuilder
84  builder = new IRBuilder<> (globalContext);
85 
86  // Build an optimizer. We're going to get default optimization settings
87  // in the same way that LLVM does.
88  FPM = new legacy::FunctionPassManager (module.get());
89 
90  FPM->add (createVerifierPass());
91 
92  PassManagerBuilder builder;
93  builder.OptLevel = 3;
94  builder.populateFunctionPassManager (*FPM);
95 
96  FPM->doInitialization ();
97 }

107 bool LLVMFormula::buildFunction ()
108 {
109  if (!engine)
110  return false;
111 
112  // First, build a prototype for the nullary function we're about to define.
113  FunctionType *FT = FunctionType::get (Type::getDoubleTy (globalContext), false);
114 
115  // Make a function
116  Function *F = Function::Create (FT, Function::ExternalLinkage, "", module.get());
117 
118  // Make a basic block to start insertion into
119  BasicBlock *BB = BasicBlock::Create (globalContext, "entry", F);
120  builder->SetInsertPoint (BB);
121 
122  // And spit the body into it
123  Value *val = (Value *)parseTree->generate (this);
124  if (val == NULL)
125  {
126  // Clean up and bail
127  F->eraseFromParent ();
128  return false;
129  }
130 
131  // Create a return statement, and check that the function makes sense
132  builder->CreateRet (val);
133 
134  // Optimize and verify
135  FPM->run (*F);
136 
137  void *fptr = engine->getPointerToFunction (F);
138  func = (FunctionPointer)(intptr_t)fptr;
139 
140  return true;
141 }

149 Constant *LLVMFormula::getGlobalVariableFor (double *ptr)
150 {
151  // Thanks to the unladen-swallow project for this snippet, which was almost
152  // impossible to figure out from the LLVM docs!
153 
154  // See if the JIT already knows about this global
155  GlobalVariable *result = const_cast<GlobalVariable *>(
156  cast_or_null<GlobalVariable>(
157  engine->getGlobalValueAtAddress (ptr)));
158  if (result && result->hasInitializer ())
159  return result;
160 
161  Constant *initializer = ConstantFP::get (Type::getDoubleTy (globalContext), *ptr);
162  if (result == NULL)
163  {
164  // Make a global variable
165  result = new GlobalVariable (*module, initializer->getType (),
166  false, GlobalVariable::InternalLinkage,
167  NULL, "");
168 
169  // Link the global variable to the right address
170  engine->addGlobalMapping (result, ptr);
171  }
172  assert (!result->hasInitializer ());
173 
174  // Add the initial value
175  result->setInitializer (initializer);
176 
177  return result;
178 }

198 void *LLVMFormula::emit (BinaryExprAST *expr)
199 {
200  if (expr->op == "=")
201  {
202  // The assign function has to be dealt with specially, we *do not*
203  // want to emit the LHS code (the load-from-memory instruction)!
204  VariableExprAST *var = dynamic_cast<VariableExprAST *>(expr->LHS);
205  if (!var) return NULL;
206 
207  Value *val = (Value *)expr->RHS->generate (this);
208  if (!val) return NULL;
209 
210  // Emit a store-to-pointer instruction
211  builder->CreateStore (val, getGlobalVariableFor (var->pointer), "storetmp");
212  return val;
213  }
214 
215  // The rest of the operators function normally
216  Value *L = (Value *)expr->LHS->generate (this);
217  Value *R = (Value *)expr->RHS->generate (this);
218  if (L == NULL || R == NULL) return NULL;
219 
220  if (expr->op == "<=")
221  {
222  L = builder->CreateFCmpULE (L, R, "letmp");
223  return builder->CreateUIToFP (L, Type::getDoubleTy (globalContext),
224  "booltmp");
225  }
226  else if (expr->op == ">=")
227  {
228  L = builder->CreateFCmpUGE (L, R, "getmp");
229  return builder->CreateUIToFP (L, Type::getDoubleTy (globalContext),
230  "booltmp");
231  }
232  else if (expr->op == "!=")
233  {
234  L = builder->CreateFCmpUNE (L, R, "neqtmp");
235  return builder->CreateUIToFP (L, Type::getDoubleTy (globalContext),
236  "booltmp");
237  }
238  else if (expr->op == "==")
239  {
240  L = builder->CreateFCmpUEQ (L, R, "eqtmp");
241  return builder->CreateUIToFP (L, Type::getDoubleTy (globalContext),
242  "booltmp");
243  }
244  else if (expr->op == "<")
245  {
246  L = builder->CreateFCmpULT (L, R, "lttmp");
247  return builder->CreateUIToFP (L, Type::getDoubleTy (globalContext),
248  "booltmp");
249  }
250  else if (expr->op == ">")
251  {
252  L = builder->CreateFCmpUGT (L, R, "gttmp");
253  return builder->CreateUIToFP (L, Type::getDoubleTy (globalContext),
254  "booltmp");
255  }
256  else if (expr->op == "+")
257  return builder->CreateFAdd (L, R, "addtmp");
258  else if (expr->op == "-")
259  return builder->CreateFSub (L, R, "subtmp");
260  else if (expr->op == "*")
261  return builder->CreateFMul (L, R, "multmp");
262  else if (expr->op == "/")
263  return builder->CreateFDiv (L, R, "divtmp");
264  else if (expr->op == "^")
265  {
266  // The floating-point intrinsics are overloaded for multiple types
267  Type *types[1] = { Type::getDoubleTy (globalContext) };
268  ArrayRef<Type *> type_array(types, 1);
269 
270  Value *func = Intrinsic::getDeclaration (module.get(), Intrinsic::pow, type_array);
271  return builder->CreateCall (func, {L, R}, "pow");
272  }
273  else
274  return NULL;
275 }

277 void *LLVMFormula::emit (CallExprAST *expr)
278 {
279  // Deal with the if-instruction first, specially
280  if (expr->function == "if")
281  {
282  Value *cond = (Value *)expr->args[0]->generate (this);
283  Value *t = (Value *)expr->args[1]->generate (this);
284  Value *f = (Value *)expr->args[2]->generate (this);
285  if (cond == NULL || t == NULL || f == NULL) return NULL;
286 
287  Value *one = ConstantFP::get (globalContext, APFloat (1.0));
288  Value *cmp = builder->CreateFCmpOEQ (cond, one, "ifcmptmp");
289 
290  return builder->CreateSelect (cmp, t, f, "ifelsetmp");
291  }
292 
293  // Sign isn't a standard-library function, implement it with a compare
294  if (expr->function == "sign")
295  {
296  Value *v = (Value *)expr->args[0]->generate (this);
297  if (!v) return NULL;
298 
299  Value *zero = ConstantFP::get (globalContext, APFloat (0.0));
300  Value *one = ConstantFP::get (globalContext, APFloat (1.0));
301  Value *none = ConstantFP::get (globalContext, APFloat (-1.0));
302 
303  Value *cmpgzero = builder->CreateFCmpOGT (v, zero, "sgncmpgzero");
304  Value *cmplzero = builder->CreateFCmpOLT (v, zero, "sgncmplzero");
305 
306  Value *fselect = builder->CreateSelect (cmplzero, none, zero, "sgnsellzero");
307  return builder->CreateSelect (cmpgzero, one, fselect, "sgnselgzero");
308  }
309 
310  //
311  // Map our function names to standard libm names
312  //
313 
314  // If we're calling "log", we want "log10"
315  if (expr->function == "log")
316  expr->function = "log10";
317  // If we're calling "ln", we want "log"
318  if (expr->function == "ln")
319  expr->function = "log";
320  // If we're calling "abs", we want "fabs"
321  if (expr->function == "abs")
322  expr->function = "fabs";
323 
324  // The following are available as LLVM intrinsics, and we should emit them
325  // specially without calling out to libm:
326  Intrinsic::ID intrinsicID = Intrinsic::not_intrinsic;
327 
328  if (expr->function == "cos")
329  intrinsicID = Intrinsic::cos;
330  else if (expr->function == "exp")
331  intrinsicID = Intrinsic::exp;
332  else if (expr->function == "log")
333  intrinsicID = Intrinsic::log;
334  else if (expr->function == "log10")
335  intrinsicID = Intrinsic::log10;
336  else if (expr->function == "sin")
337  intrinsicID = Intrinsic::sin;
338  else if (expr->function == "sqrt")
339  intrinsicID = Intrinsic::sqrt;
340 
341  if (intrinsicID != Intrinsic::not_intrinsic)
342  {
343  // Most of the floating-point intrinsics are overloaded for multiple types
344  Type *types[1] = { Type::getDoubleTy (globalContext) };
345  ArrayRef<Type *> type_array(types, 1);
346 
347  Value *arg = (Value *)expr->args[0]->generate (this);
348  if (!arg) return NULL;
349 
350  Value *func = Intrinsic::getDeclaration (module.get(), intrinsicID, type_array);
351  return builder->CreateCall (func, arg, expr->function);
352  }
353 
354  // The rest of these are calls to stdlib floating point math
355  // functions.
356 
357  // atan2 is special, because it has two arguments
358  if (expr->function == "atan2")
359  {
360  Value *arg1 = (Value *)expr->args[0]->generate (this);
361  Value *arg2 = (Value *)expr->args[1]->generate (this);
362  if (!arg1 || !arg2) return NULL;
363 
364  Module *M = builder->GetInsertBlock ()->getParent ()->getParent ();
365  Value *Callee = M->getOrInsertFunction (expr->function,
366  Type::getDoubleTy (globalContext),
367  Type::getDoubleTy (globalContext),
368  Type::getDoubleTy (globalContext));
369  return builder->CreateCall (Callee, {arg1, arg2}, expr->function);
370  }
371  else
372  {
373  // Emit a call to function (arg)
374  Value *arg = (Value *)expr->args[0]->generate (this);
375  if (!arg) return NULL;
376 
377  Module *M = builder->GetInsertBlock ()->getParent ()->getParent ();
378  Value *Callee = M->getOrInsertFunction (expr->function,
379  Type::getDoubleTy (globalContext),
380  Type::getDoubleTy (globalContext));
381  return builder->CreateCall (Callee, arg, expr->function);
382  }
383 }
