57 void *LibJITFormula::emit (BinaryExprAST *expr)
58 {
59  // Deal with assign separately, we don't want to compile the LHS
60  if (expr->op == "=")
61  {
62  VariableExprAST *var = dynamic_cast<VariableExprAST *>(expr->LHS);
63  if (!var) return NULL;
64 
65  jit_value_t val = (jit_value_t)expr->RHS->generate (this);
66  if (!val) return NULL;
67 
68  // Get a pointer for this variable
69  jit_value_t pointer = jit_value_create_nint_constant (function, jit_type_void_ptr,
70  (jit_nint)var->pointer);
71 
72  // Emit a store instruction
73  jit_insn_store_relative (function, pointer, 0, val);
74 
75  // Return value
76  return val;
77  }
78 
79  // Generate both sides
80  jit_value_t L = (jit_value_t)expr->LHS->generate (this);
81  jit_value_t R = (jit_value_t)expr->RHS->generate (this);
82  if (L == NULL || R == NULL) return NULL;
83 
84  if (expr->op == "<=")
85  return jit_insn_le (function, L, R);
86  else if (expr->op == ">=")
87  return jit_insn_ge (function, L, R);
88  else if (expr->op == "!=")
89  return jit_insn_ne (function, L, R);
90  else if (expr->op == "==")
91  return jit_insn_eq (function, L, R);
92  else if (expr->op == "<")
93  return jit_insn_lt (function, L, R);
94  else if (expr->op == ">")
95  return jit_insn_gt (function, L, R);
96  else if (expr->op == "+")
97  return jit_insn_add (function, L, R);
98  else if (expr->op == "-")
99  return jit_insn_sub (function, L, R);
100  else if (expr->op == "*")
101  return jit_insn_mul (function, L, R);
102  else if (expr->op == "/")
103  return jit_insn_div (function, L, R);
104  else if (expr->op == "^")
105  {
106  // jit_insn_pow seems to give the wrong results?
107  jit_type_t params[2];
108  params[0] = jit_type_float64;
109  params[1] = jit_type_float64;
110  jit_type_t signature;
111  signature = jit_type_create_signature (jit_abi_cdecl, jit_type_float64,
112  params, 2, 1);
113  jit_value_t args[2];
114  args[0] = L;
115  args[1] = R;
116 
117  return jit_insn_call_native (function, "pow", (void *)pow, signature,
118  args, 2, 0);
119  }
120  else
121  return NULL;
122 }

124 void *LibJITFormula::emit (CallExprAST *expr)
125 {
126  if (expr->function == "if")
127  {
128  // We're going to make a new temporary, and store a value to it depending
129  // on what happens here.
130  jit_value_t result = jit_value_create (function, jit_type_float64);
131 
132  // if (cond)
133  jit_value_t cond = (jit_value_t)expr->args[0]->generate (this);
134  if (!cond) return NULL;
135 
136  jit_value_t one = jit_value_create_float64_constant (function, jit_type_float64,
137  1.0);
138  jit_value_t comparison = jit_insn_eq (function, cond, one);
139  jit_label_t label_if = jit_label_undefined;
140  jit_label_t label_end = jit_label_undefined;
141 
142  jit_insn_branch_if_not (function, comparison, &label_if);
143 
144  // (the if-value)
145  jit_value_t t = (jit_value_t)expr->args[1]->generate (this);
146  jit_insn_store (function, result, t);
147  jit_insn_branch (function, &label_end);
148 
149  // The else branches to here...
150  jit_insn_label (function, &label_if);
151 
152  // (the else-value)
153  jit_value_t f = (jit_value_t)expr->args[2]->generate (this);
154  jit_insn_store (function, result, f);
155 
156  jit_insn_label (function, &label_end);
157 
158  return result;
159  }
160  else if (expr->function == "sign")
161  {
162  // Same thing as the if-function above
163  jit_value_t one = jit_value_create_float64_constant (function, jit_type_float64,
164  1.0);
165  jit_value_t zero = jit_value_create_float64_constant (function, jit_type_float64,
166  0.0);
167  jit_value_t minusone = jit_value_create_float64_constant (function,
168  jit_type_float64,
169  -1.0);
170 
171  jit_value_t arg = (jit_value_t)expr->args[0]->generate (this);
172  jit_value_t result = jit_value_create (function, jit_type_float64);
173  jit_label_t label_end = jit_label_undefined;
174 
175  jit_value_t positive = jit_insn_gt (function, arg, zero);
176  jit_label_t label_positive = jit_label_undefined;
177  jit_insn_branch_if_not (function, positive, &label_positive);
178  jit_insn_store (function, result, one);
179  jit_insn_branch (function, &label_end);
180  jit_insn_label (function, &label_positive);
181 
182  jit_value_t negative = jit_insn_lt (function, arg, zero);
183  jit_label_t label_negative = jit_label_undefined;
184  jit_insn_branch_if_not (function, negative, &label_negative);
185  jit_insn_store (function, result, minusone);
186  jit_insn_branch (function, &label_end);
187  jit_insn_label (function, &label_negative);
188 
189  jit_insn_store (function, result, zero);
190  jit_insn_label (function, &label_end);
191 
192  return result;
193  }
194  else if (expr->function == "atan2")
195  {
196  // HACK: We have to put this here since it has two arguments
197  jit_value_t args[2];
198  args[0] = (jit_value_t)expr->args[0]->generate (this);
199  args[1] = (jit_value_t)expr->args[1]->generate (this);
200 
201  jit_type_t params[2];
202  params[0] = jit_type_float64;
203  params[1] = jit_type_float64;
204  jit_type_t signature;
205  signature = jit_type_create_signature (jit_abi_cdecl, jit_type_float64,
206  params, 2, 1);
207 
208  return jit_insn_call_native (function, "atan2", (void *)atan2, signature,
209  args, 2, 0);
210  }
211 
212  // Compile the arguments
213  // HACK: we just assume for now that if() is the only thing with
214  // more than one argument.
215  jit_value_t arg = (jit_value_t)expr->args[0]->generate (this);
216 
217  // Create a native signature for our few native functions
218  jit_type_t params[1];
219  params[0] = jit_type_float64;
220  jit_type_t signature;
221  signature = jit_type_create_signature (jit_abi_cdecl, jit_type_float64,
222  params, 1, 1);
223 
224  // Several of the jit_insn_FUNC constructs below have been replaced
225  // by jit_insn_call_native, because the intrinsic instructions seem to
226  // give the wrong answers.
227  if (expr->function == "sin")
228  return jit_insn_call_native (function, "sin", (void *)sin, signature,
229  &arg, 1, 0);
230  else if (expr->function == "cos")
231  return jit_insn_call_native (function, "cos", (void *)cos, signature,
232  &arg, 1, 0);
233  else if (expr->function == "tan")
234  return jit_insn_call_native (function, "tan", (void *)tan, signature,
235  &arg, 1, 0);
236  else if (expr->function == "asin")
237  return jit_insn_call_native (function, "asin", (void *)asin, signature,
238  &arg, 1, 0);
239  else if (expr->function == "acos")
240  return jit_insn_call_native (function, "acos", (void *)acos, signature,
241  &arg, 1, 0);
242  else if (expr->function == "atan")
243  return jit_insn_call_native (function, "atan", (void *)atan, signature,
244  &arg, 1, 0);
245  else if (expr->function == "sinh")
246  return jit_insn_call_native (function, "sinh", (void *)sinh, signature,
247  &arg, 1, 0);
248  else if (expr->function == "cosh")
249  return jit_insn_call_native (function, "cosh", (void *)cosh, signature,
250  &arg, 1, 0);
251  else if (expr->function == "tanh")
252  return jit_insn_call_native (function, "tanh", (void *)tanh, signature,
253  &arg, 1, 0);
254  else if (expr->function == "asinh")
255  return jit_insn_call_native (function, "asinh", (void *)asinh, signature,
256  &arg, 1, 0);
257  else if (expr->function == "acosh")
258  return jit_insn_call_native (function, "acosh", (void *)acosh, signature,
259  &arg, 1, 0);
260  else if (expr->function == "atanh")
261  return jit_insn_call_native (function, "atanh", (void *)atanh, signature,
262  &arg, 1, 0);
263  else if (expr->function == "log2")
264  return jit_insn_call_native (function, "log2", (void *)log2, signature,
265  &arg, 1, 0);
266  else if (expr->function == "log" || expr->function == "log10")
267  return jit_insn_call_native (function, "log10", (void *)log10, signature,
268  &arg, 1, 0);

287 bool LibJITFormula::buildFunction ()
288 {
289  // Create a context
290  jit_context_t context;
291  context = jit_context_create ();
292  jit_context_build_start (context);
293 
294  // Create a signature for double (*func)();
295  jit_type_t signature;
296  signature = jit_type_create_signature (jit_abi_cdecl, jit_type_float64,
297  NULL, 0, 1);
298 
299  // Create a function
300  function = jit_function_create (context, signature);
301 
302  // Return the result of compiling the whole mess
303  jit_value_t val;
304  val = (jit_value_t)parseTree->generate (this);
305  if (!val)
306  return false;
307 
308  jit_insn_return (function, val);
309  jit_context_build_end (context);
310 
311  if (!jit_function_compile (function))
312  {
313  jit_function_abandon (function);
314  return false;
315  }
316 
317  func = (FunctionPointer)jit_function_to_closure (function);
318  if (!func)
319  {
320  jit_function_abandon (function);
321  return false;
322  }
323 
324  return true;
325 }
